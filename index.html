#include #include #include #include #include #include // <--- NUEVA LIBRERÍA NECESARIA #include "PaginaWeb.h" // --- CONFIGURACIÓN DE HARDWARE --- #define DHTPIN 4 #define DHTTYPE DHT11 DHT dht(DHTPIN, DHTTYPE); // --- CONFIGURACIÓN DEL LED RGB --- #define PIN_ROJO 14 #define PIN_VERDE 12 #define PIN_AZUL 13 // --- CONFIGURACIÓN DEL SERVO --- #define PIN_SERVO 15 // Conecta el cable naranja/amarillo del servo a este pin Servo motorBarrido; bool servoActivo = false; int posServo = 0; int direccionServo = 5; // Aumenta este número si quieres que gire aún más rápido unsigned long lastServo = 0; // --- CREDENCIALES --- const char* ssid = "NauticaNet"; const char* password = "PromoXXX.2026"; const char* mqtt_server = "df734b8fbeed43978f29869442892dcf.s1.eu.hivemq.cloud"; const int mqtt_port = 8883; const char* mqtt_user = "Admin"; const char* mqtt_pass = "FluxaIgnis2026"; const char* topic_publish = "fxi/datos"; // --- CLIENTES --- WiFiClientSecure espClient; PubSubClient client(espClient); WebServer server(80); unsigned long lastMsg = 0; // --- FUNCIONES DE CONTROL RGB --- void setColor(int r, int g, int b) { // Al restar "255 - valor" corregimos el problema del Ánodo Común // Ahora 0 = apagado y 255 = máximo brillo analogWrite(PIN_ROJO, 255 - r); analogWrite(PIN_VERDE, 255 - g); analogWrite(PIN_AZUL, 255 - b); } void actualizarLedAmbiental(float temp) { // Rango ajustado: 26°C (estable) a 40°C (calor intenso) float tempConstrained = constrain(temp, 26.0, 40.0); // A 26°C -> Azul total. A 40°C -> Rojo total. int rojo = map(tempConstrained, 26, 40, 0, 255); int azul = map(tempConstrained, 26, 40, 255, 0); int verde = 0; setColor(rojo, verde, azul); } // --- FUNCIONES WEB --- void handleRoot() { server.send(200, "text/html", index_html); } void handleDatos() { float h = dht.readHumidity(); float t = dht.readTemperature(); long rssi = WiFi.RSSI(); String json = "{\"temp\":" + String(t) + ",\"hum\":" + String(h) + ",\"rssi\":" + String(rssi) + "}"; server.send(200, "application/json", json); } // Nueva ruta para el botón del Servo void handleToggleServo() { servoActivo = !servoActivo; server.send(200, "text/plain", servoActivo ? "ON" : "OFF"); } void setup() { Serial.begin(115200); pinMode(PIN_ROJO, OUTPUT); pinMode(PIN_VERDE, OUTPUT); pinMode(PIN_AZUL, OUTPUT); setColor(0, 0, 0); setup_wifi(); dht.begin(); // Inicializar Servo motorBarrido.setPeriodHertz(50); motorBarrido.attach(PIN_SERVO); espClient.setInsecure(); client.setServer(mqtt_server, mqtt_port); server.on("/", handleRoot); server.on("/datos", handleDatos); server.on("/toggleServo", handleToggleServo); // Ruta añadida server.begin(); Serial.println("🔥 FXI System: Servidor Híbrido Iniciado"); } void setup_wifi() { delay(10); Serial.print("\nConectando a " + String(ssid)); WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\n¡Conectado! IP Local: " + WiFi.localIP().toString()); } void reconnect() { while (!client.connected()) { Serial.print("Intentando MQTT..."); String clientId = "ESP32_FXI_Client-" + String(WiFi.macAddress()); if (client.connect(clientId.c_str(), mqtt_user, mqtt_pass)) { Serial.println("Conectado a FXI Cloud"); } else { Serial.print("Error rc="); Serial.print(client.state()); delay(5000); } } } void loop() { server.handleClient(); if (!client.connected()) { reconnect(); } client.loop(); unsigned long now = millis(); // --- BARRIDO FLUIDO DEL SERVO --- if (servoActivo) { if (now - lastServo > 15) { // 15 milisegundos = movimiento súper fluido lastServo = now; posServo += direccionServo; // Control de los límites del giro if (posServo >= 180) { posServo = 180; direccionServo = -direccionServo; // Rebota hacia atrás } else if (posServo <= 0) { posServo = 0; direccionServo = -direccionServo; // Rebota hacia adelante } motorBarrido.write(posServo); } } // --- LECTURA DEL DHT11 --- if (now - lastMsg > 1000) { lastMsg = now; float h = dht.readHumidity(); float t = dht.readTemperature(); long rssi = WiFi.RSSI(); if (!isnan(h) && !isnan(t)) { actualizarLedAmbiental(t); String payload = "{\"temp\":" + String(t) + ",\"hum\":" + String(h) + ",\"rssi\":" + String(rssi) + "}"; client.publish(topic_publish, payload.c_str()); Serial.print("Telemetría: "); Serial.println(payload); } else { Serial.println("⚠️ Error en DHT11"); setColor(0, 0, 0); } } }
